-- Church booleans
T    = /\X. \x:X. \y:X. x
F    = /\X. \x:X. \y:X. y

and  = \x:forall X. X -> X -> X. \y:forall X. X -> X -> X. x [forall X. X -> X -> X] y F
or   = \x:forall X. X -> X -> X. \y:forall X. X -> X -> X. x [forall X. X -> X -> X] T y
not  = \x:forall X. X -> X -> X. x [forall X. X -> X -> X] F T

-- Church numerals
0    = /\X. \x:X. \f:X -> X. x
succ = \n:forall X. (X -> X) -> X -> X. /\X. \f:X -> X. \x:X. f (n [X] f x)
