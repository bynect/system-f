-- Church booleans
bool  = [forall a. a -> a -> a]
true  = /\a. \x:a. \y:a. x
false = /\a. \x:a. \y:a. y

-- Church numerals
nat   = [forall a. a -> (a -> a) -> a]
zero  = /\a. \x:a. \f:a -> a. x
succ  = \n:nat. /\a. \x:a. \f:a -> a. f (n [a] x f)
